getDate() = Dia do Mês
getDay() = Dia da semana (número)
getFullYear() = Ano com 4 dígitos
getHours() = Horas
getMinutes() = Minutos
getSeconds() = Segundos
getMilliseconds() = Milisegundos

Date.now() = Timestamp (milisegundos desde 1 de janeiro de 1970, 00:00:00 UTC)

getTimezoneOffset() = Timezone da localidade
